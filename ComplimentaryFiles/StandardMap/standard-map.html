def plot_map(r, x0, n):
    t = np.linspace(0, 1)
    plt.plot(t, logistic(r, t), 'k', lw=2)
    plt.plot([0, 1], [0, 1], 'k', lw=2)
    x = x0
    for i in range(n):
        y = logistic(r, x)
        plt.plot([x, x], [x, y], 'k', lw=1)
        plt.plot([x, y], [y, y], 'k', lw=1)
        plt.plot([x], [y], 'ok', ms=10, alpha=(i + 1) / n)
        x = y
        
plot_map(2.5, .1, 10)
